# Grammar Definition
Expression ::= Term (('+' | '-') Term)*
Term ::= Factor (('*' | '/') Factor)*
Factor ::= Number | '(' Expression ')'
Number ::= [0-9]+

# Python Implementation
parser = PEGParser()
result = parser.parse("2 + 3 * 4")
